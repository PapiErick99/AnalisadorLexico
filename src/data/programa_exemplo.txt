int main() {
    int &x, &y;
    float &z;

    &x = 10;
    &y = 20;
    &z = &x + &y * 2;

    ifs (&x < &y) {
        &x = &x + 1;
    } else {
        &x = &x - 1;
    }

    while (&x < 30) {
        &x = &x + 1;
    }

    do {
        &y = &y - 1;
    } while (&y > 0);

    for (&x = 0; &x < 10; &x = &x + 1) {
        &z = &z + &x;
    }
}
